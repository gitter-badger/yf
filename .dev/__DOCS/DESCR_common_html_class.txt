//*******************************************************************************************************//
function select_box()
	параметры:
		-$name			- (string) аттрибут "name" html-тега "select"
		-$values		- (array) массив значений для отображения в выпадающем списке
		-$selected		- (string) выбранное значение для выпадающего списка
		-$show_text		- (bool) показывать или нет приглашение "--select ...--" если ничего не выбрано 
(default:true)
		-$type			- (string 1 or 2) определяет из ключей или значений массива "$values" выбирать 
значение "$selected" (1 - значение, 2 - ключ) (default:2)
		-$add_str		- (string) с помощью этого параметра передаются аттрибуты в тело тега "select" 
(кроме name)
		-$translate		- (bool) определяет переводить или нет переданные значения (default:false)
		-$level			- ???
	возвращает - html-код выпадающего списка (select_box).

function radio_box ()
	параметры:
		- $box_name			- (string) аттрибут "name" html-тега input [type = radio]
		- $values			- (array) массив значений для отображения
		- $selected 		- (string) выбранное значение 
		- $flow_vertical 	- (bool) определяет отображать значения вертикально или нет(default:false) 
		- $type 			- (string 1 or 2) определяет из ключей или значений массива "$values" выбирать 
значение "$selected" (1 - значение, 2 - ключ) (default:2) 
		- $add_str 			- (string) с помощью этого параметра передаются аттрибуты в тело тега (кроме name) 
		- $translate 		- (bool) определяет переводить или нет переданные значения (default:false)

	возвращает - html-код элемента input [type = radio]

function date_box()
	параметры:
		- $selected_date 	- (int) дата, которая будет отображаться как выбранная
		- $years			- (string) диапазон годов для отображения ("начальный"-"конечный")
		- $name_postfix 	- (string) суффикс в аттрибутах "name" выпадающих списков (например "_birth" - "year_birth",
"month_birth", "day_birth")
		- $add_str 			- (string) с помощью этого параметра передаются аттрибуты в тело тега "select" 
(кроме name) (в функции не подключено!!!)
		- $order 			- (string) порядок отображения (default:ymd) 
		- $show_text 		- (bool) показывать или нет приглашение "--select ...--" если ничего не выбрано 
(default:true)

	возвращает - html-код 3-х выпадающих списков (день, месяц , год)
